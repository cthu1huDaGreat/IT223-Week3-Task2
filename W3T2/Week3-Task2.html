SELECT e.employee_id, CONCAT(e.first_name,' ',e.last_name) AS 'name', j.job_title FROM employees AS e INNER JOIN jobs j ON j.job_id=e.job_id

CREATE OR REPLACE VIEW view_address AS SELECT CONCAT(1.street_address,' ',l.postal_code,' ',l.city,' ',l.state_province,' ') AS 'address', c.country_name, r.region_name FROM locations AS l INNER JOIN countries c ON l.country_id=c.country_id INNER JOIN regions r ON r.region_id=c.region_id;

CREATE OR REPLACE VIEW emp_view_jobs AS SELECT e.employee_id, CONCAT(e.first_name,' ',e.last_name) AS 'name', j.job_title FROM employees AS e INNER JOIN jobs j ON j.job_id=e.job_id;
